```bash
#!/bin/bash
# ── NERVE Deploy Script ─────────────────────────────────────────────────────
# Usage: ./deploy.sh
# Deploys the latest main branch to the production VPS.
# Prerequisites: SSH key auth configured for VPS_HOST.

set -e

VPS_HOST="root@YOUR_VPS_IP"   # ← Replace with your Hetzner VPS IP before first use
APP_DIR="/opt/nerve/app"
VENV_DIR="/opt/nerve/venv"

echo "[deploy] Connecting to $VPS_HOST..."

ssh "$VPS_HOST" bash -s << EOF
  set -e
  echo "[deploy] Pulling latest code..."
  cd $APP_DIR
  git pull origin main

  echo "[deploy] Installing dependencies..."
  $VENV_DIR/bin/pip install -r requirements.txt --quiet

  echo "[deploy] Restarting nerve service..."
  sudo systemctl restart nerve

  echo "[deploy] Service status:"
  sudo systemctl status nerve --no-pager -l
EOF

echo "[deploy] Done."
```

**deploy/nginx.conf** — Per D-01 and D-08. Full nginx configuration for nerve.app with HTTP redirect, HTTPS termination, SSL via Let's Encrypt, and WebSocket upgrade headers for Socket.IO:

```nginx
# ── NERVE nginx Configuration ────────────────────────────────────────────────
# Domain: nerve.app
# Copy to: /etc/nginx/sites-available/nerve
# Enable: sudo ln -s /etc/nginx/sites-available/nerve /etc/nginx/sites-enabled/nerve
# Reload: sudo nginx -t && sudo systemctl reload nginx
# SSL certs: Run certbot --nginx -d nerve.app -d www.nerve.app first

# ── HTTP → HTTPS redirect ────────────────────────────────────────────────────
server {
    listen 80;
    listen [::]:80;
    server_name nerve.app www.nerve.app;

    location / {
        return 301 https://nerve.app$request_uri;
    }
}

# ── HTTPS server ─────────────────────────────────────────────────────────────
server {
    listen 443 ssl;
    listen [::]:443 ssl;
    server_name nerve.app www.nerve.app;

    # SSL — managed by certbot
    ssl_certificate     /etc/letsencrypt/live/nerve.app/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/nerve.app/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    # Redirect www to apex
    if ($host = www.nerve.app) {
        return 301 https://nerve.app$request_uri;
    }

    # ── Proxy to gunicorn ────────────────────────────────────────────────────
    location / {
        proxy_pass         http://127.0.0.1:8000;
        proxy_http_version 1.1;

        # WebSocket upgrade headers (required for Socket.IO — D-08)
        proxy_set_header Upgrade    $http_upgrade;
        proxy_set_header Connection "upgrade";

        # Standard proxy headers
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Timeouts — long-lived WebSocket connections need generous values
        proxy_read_timeout  3600s;
        proxy_send_timeout  3600s;
        proxy_connect_timeout 60s;
    }
}
```

**deploy/nerve.service** — Per D-03. systemd unit file for gunicorn with gthread worker class:

```ini
# ── NERVE systemd Service Unit ───────────────────────────────────────────────
# Copy to: /etc/systemd/system/nerve.service
# Enable:  sudo systemctl daemon-reload && sudo systemctl enable nerve
# Start:   sudo systemctl start nerve
# Logs:    sudo journalctl -u nerve -f

[Unit]
Description=NERVE Sales Assistant (gunicorn)
After=network.target

[Service]
Type=simple
User=www-data
Group=www-data
WorkingDirectory=/opt/nerve/app

# Environment — secrets live in /etc/nerve/.env (not in repo)
EnvironmentFile=/etc/nerve/.env

# gunicorn with gthread worker class (required for Flask-SocketIO threading mode)
# 1 worker + 4 threads: optimal for CX22 (2 vCPU, 4 GB RAM) with Socket.IO
ExecStart=/opt/nerve/venv/bin/gunicorn \
    --worker-class gthread \
    --workers 1 \
    --threads 4 \
    --bind 127.0.0.1:8000 \
    --timeout 120 \
    --access-logfile - \
    --error-logfile - \
    app:app

# Restart policy
Restart=always
RestartSec=5s

# Security hardening
NoNewPrivileges=true
PrivateTmp=true

[Install]
WantedBy=multi-user.target
```

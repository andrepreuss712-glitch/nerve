let selectedPersonalityTypeId = null;
let generatedPersonality = null;  // holds temp generated data before save

async function loadPersonalityTypes() {
    try {
        const res = await fetch('/api/training/personalities');
        const types = await res.json();
        const container = document.getElementById('t-typ-cards');
        container.innerHTML = types.map(t => `
            <div class="t-typ-card" data-id="${t.id}" onclick="selectTyp(${t.id}, this)"
                 style="background:#1C2333; border:1px solid #2D3748; border-radius:12px; padding:16px; cursor:pointer; text-align:center; transition:border-color 0.2s;">
                <div style="font-size:32px; margin-bottom:8px;">${t.icon || '👤'}</div>
                <div style="color:#E2E8F0; font-weight:600; font-size:14px;">${t.name}</div>
                <div style="color:#A0AEC0; font-size:12px; margin-top:4px;">${t.kurzbeschreibung || ''}</div>
                ${t.is_custom ? '<div style="color:#00D4AA; font-size:10px; margin-top:4px;">Eigener Typ</div>' : ''}
            </div>
        `).join('');
    } catch(e) {
        console.error('[Training] Failed to load personality types:', e);
    }
}

function selectTyp(id, el) {
    selectedPersonalityTypeId = id;
    generatedPersonality = null;
    document.querySelectorAll('.t-typ-card').forEach(c => c.style.borderColor = '#2D3748');
    if (el) el.style.borderColor = '#00D4AA';
    document.getElementById('t-typ-briefing').style.display = 'none';
    // Advance to difficulty
    showSection('t-diff-section');
}

async function generateRandomPersonality() {
    const btn = document.getElementById('t-typ-random-btn');
    btn.disabled = true;
    btn.textContent = 'Generiere...';
    try {
        const res = await fetch('/api/training/personalities/generate', {method: 'POST'});
        const data = await res.json();
        if (data.error) throw new Error(data.error);
        generatedPersonality = data;
        selectedPersonalityTypeId = null;  // not saved yet, no ID
        
        // Show briefing card (D-02)
        document.getElementById('t-typ-briefing').style.display = 'block';
        document.getElementById('t-typ-briefing-text').textContent = data.briefing || data.kurzbeschreibung || '';
        
        // Deselect any card
        document.querySelectorAll('.t-typ-card').forEach(c => c.style.borderColor = '#2D3748');
    } catch(e) {
        console.error('[Training] Generation failed:', e);
        alert('Generierung fehlgeschlagen. Bitte erneut versuchen.');
    } finally {
        btn.disabled = false;
        btn.textContent = '\u{1F3B2} Zufaellig generieren';
    }
}

function confirmGeneratedPersonality() {
    // User confirmed generated personality, advance to difficulty
    showSection('t-diff-section');
}
```

function updateMoodIndicator(stimmung) {
    const indicator = document.getElementById('t-mood-indicator');
    if (stimmung === null || stimmung === undefined) {
        indicator.style.display = 'none';
        return;
    }
    indicator.style.display = 'flex';
    const fill = document.getElementById('t-mood-fill');
    const value = document.getElementById('t-mood-value');
    
    // Map -5..+5 to 0..100%
    const pct = ((stimmung + 5) / 10) * 100;
    fill.style.width = pct + '%';
    
    // Color: red < -2, orange -2..0, yellow 1..3, green 4..5
    let color = '#48BB78';  // green
    if (stimmung <= -3) color = '#FC8181';       // red
    else if (stimmung <= 0) color = '#F6AD55';   // orange
    else if (stimmung <= 3) color = '#F6E05E';   // yellow
    
    fill.style.backgroundColor = color;
    value.textContent = stimmung > 0 ? '+' + stimmung : stimmung;
    value.style.color = color;
}
```

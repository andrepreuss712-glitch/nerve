```javascript
function renderEwbButtons() {
  const bar = document.getElementById('ewbBar');
  if (!bar) return;

  // Get einwaende from profile or fallback
  let einwaende = DACH_FALLBACK_EINWAENDE;
  try {
    const pd = window._profileEinwaende || {};
    if (pd.einwaende && pd.einwaende.length > 0) {

    }
  } catch(e) {}
  if (!einwaende.length) einwaende = DACH_FALLBACK_EINWAENDE;

  bar.innerHTML = einwaende.map(typ =>
    `<button class="ewb-btn" onclick="triggerEwb('${escHtml(typ)}')" title="Einwand: ${escHtml(typ)}">🛡️ ${escHtml(typ)}</button>`
  ).join('');
}
```

**Replace with:**
```javascript
function renderEwbButtons() {
  const bar    = document.getElementById('ewbBar');
  const kpBar  = document.getElementById('kp-ewbBar');
  if (!bar && !kpBar) return;

  // Get einwaende from profile or fallback
  let einwaende = DACH_FALLBACK_EINWAENDE;
  try {
    const pd = window._profileEinwaende || {};
    if (pd.einwaende && pd.einwaende.length > 0) {
      einwaende = pd.einwaende.map(e => e.typ || e.name || e).filter(Boolean);
    }
  } catch(e) {}
  if (!einwaende.length) einwaende = DACH_FALLBACK_EINWAENDE;

  const html = einwaende.map(typ =>
    `<button class="ewb-btn" onclick="triggerEwb('${escHtml(typ)}')" title="Einwand: ${escHtml(typ)}">\uD83D\uDEE1\uFE0F ${escHtml(typ)}</button>`
  ).join('');

  if (bar)   bar.innerHTML   = html;
  if (kpBar) kpBar.innerHTML = html;
}
```

Read the file first and match exactly.

The `triggerEwb()` function does NOT need changes — it queries `.ewb-btn` by text content which works for both bars, and it renders the response in `#ai` (the full-mode AI panel). This is acceptable; in compact mode the user typically switches back to full mode to read the response. No changes to `triggerEwb()`.

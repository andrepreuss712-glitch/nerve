```python
# ── Phase 04.8: Readiness score + active hint orchestration ──
from services.ki_logik import compute_readiness_score, select_active_hint, dynamic_ewb_buttons

with ls.state_lock:
    factors = ls.state.get('score_factors_seen', {}) or {}
    ergebnis = ls.state.get('ergebnis') or {}
    cur_phase = ls.state.get('current_phase', 1)
    cur_phase_name = ls.state.get('current_phase_name', 'Opener')
    mode = ls.state.get('mode', 'meeting')
    cold_inf = ls.state.get('cold_call_inference')

# Update factor tally from latest ergebnis (non-destructive — increments only)
# ergebnis may carry signal flags from analysiere_mit_claude
if ergebnis:
    if ergebnis.get('kaufsignal'):
        factors['kaufsignal'] = factors.get('kaufsignal', 0) + 1
    if ergebnis.get('einwand') and ergebnis.get('einwand_geloest'):
        factors['einwand_geloest'] = factors.get('einwand_geloest', 0) + 1
    elif ergebnis.get('einwand'):
        factors['einwand_offen'] = factors.get('einwand_offen', 0) + 1
    for key in ('detailfrage','budget_erwaehnt','naechster_schritt','zustimmung','konkurrenz','zeitdruck_kunde','monosyllabisch'):
        if ergebnis.get(key):
            factors[key] = factors.get(key, 0) + 1

score, bucket = compute_readiness_score({'score_factors_seen': factors}, [])

# Build hint candidates
candidates = []
if ergebnis.get('kritischer_fehler'):
    candidates.append({'type':'critical','priority':1,'text':ergebnis.get('kritischer_fehler')[:120],'color':'red','source':'analyse_loop','ts':datetime.utcnow().isoformat()})
if bucket == 'closing' and ergebnis.get('kb_delta', 0) > 0:
    candidates.append({'type':'kaufsignal','priority':2,'text':'Kaufsignal erkannt — jetzt abschließen','color':'gold','source':'readiness_rule','ts':datetime.utcnow().isoformat()})
if ergebnis.get('einwand') and ergebnis.get('gegenargument'):
    candidates.append({'type':'einwand','priority':3,'text':ergebnis.get('gegenargument')[:120],'color':'orange','source':'analyse_loop','ts':datetime.utcnow().isoformat()})
if cold_inf and cold_inf.get('recommended_next'):
    candidates.append({'type':'phase','priority':4,'text':cold_inf['recommended_next'][:120],'color':'blue','source':'cold_call_infer','ts':datetime.utcnow().isoformat()})
if ergebnis.get('tipp'):
    candidates.append({'type':'tipp','priority':5,'text':ergebnis.get('tipp')[:120],'color':'gray','source':'analyse_loop','ts':datetime.utcnow().isoformat()})

active_hint = select_active_hint(candidates)

# Dynamic EWB buttons for current phase (fallback to profile buttons if phase unknown)
try:
    base_buttons = ls.get_active_profile_ewbs() if hasattr(ls,'get_active_profile_ewbs') else None
except Exception:
    base_buttons = None
ewb_buttons = dynamic_ewb_buttons(cur_phase, base_buttons)

with ls.state_lock:
    ls.state['score_factors_seen'] = factors
    ls.state['readiness_score'] = score
    ls.state['readiness_bucket'] = bucket
    ls.state['kaufbereitschaft'] = score   # legacy mirror per RESEARCH Q2 R2
    ls.state['active_hint'] = active_hint
    ls.state['ewb_buttons'] = ewb_buttons

ls.kaufbereitschaft = score  # module global mirror
```

If `ls.get_active_profile_ewbs` doesn't exist, inline a safe attribute access to the active profile dict's einwand button list, with empty fallback. Do NOT break if profile missing.

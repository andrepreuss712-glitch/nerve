class ApiCostLog(Base):
    """Jeder einzelne API-Call mit eingefrorenem Wechselkurs und gefrorener Rate.
    D-02: Wechselkurs wird beim Schreiben eingefroren (steuerlich korrekt).
    """
    __tablename__ = 'api_cost_log'
    id = Column(Integer, primary_key=True)
    created_at = Column(DateTime, default=utcnow, nullable=False, index=True)
    provider = Column(String(32), nullable=False, index=True)  # 'anthropic' | 'deepgram' | 'elevenlabs' | 'stripe'
    model = Column(String(64), nullable=False)                 # 'haiku-4-5' | 'nova-2' | 'multilingual-v2' | ...
    user_id = Column(Integer, ForeignKey('users.id'), nullable=True, index=True)
    org_id = Column(Integer, ForeignKey('organisations.id'), nullable=True, index=True)
    units = Column(Numeric(14, 4), nullable=False)             # tokens/1000 | minutes | chars/1000
    unit_type = Column(String(32), nullable=False)             # 'per_1k_input_tokens' | 'per_1k_output_tokens' | 'per_minute' | 'per_1k_chars'
    rate_applied = Column(Numeric(12, 8), nullable=False)      # Preis zum Zeitpunkt des Calls in Rate-Currency
    rate_currency = Column(String(3), nullable=False, default='USD')
    fx_rate_applied = Column(Numeric(10, 6), nullable=False)   # USD->EUR Kurs zum Zeitpunkt
    cost_eur = Column(Numeric(12, 6), nullable=False)          # Endbetrag in EUR (abgeleitet, gespeichert für schnelle SUMs)
    session_id = Column(String(64), nullable=True, index=True) # optional Korrelation zu ConversationLog
    context_tag = Column(String(32), nullable=True)            # 'live_haiku' | 'post_sonnet' | 'ewb_rank' | 'training_tts' | 'stt'

class ApiRate(Base):

    last_checked_at = Column(DateTime, default=utcnow, nullable=False)  # D-06 Preismonitor
    source_url = Column(String(512), nullable=True)
    created_at = Column(DateTime, default=utcnow, nullable=False)

class PriceChangeLog(Base):

    changed_at = Column(DateTime, default=utcnow, nullable=False, index=True)

    new_rate = Column(Numeric(12, 8), nullable=False)

class FixedCost(Base):

class RevenueLog(Base):

class ExchangeRate(Base):

    created_at = Column(DateTime, default=utcnow, nullable=False)

```python
def _seed_founder_dashboard_defaults():
    """Phase 04.7.2 — idempotenter Seed von FixedCost + ApiRate + ExchangeRate-Fallback.
    Re-run-safe: prüft COUNT pro Tabelle vor INSERT.
    """
    from database.db import SessionLocal
    from database.models import FixedCost, ApiRate, ExchangeRate
    from datetime import date
    db = SessionLocal()
    try:
        # --- FixedCost Seed (D-10 Briefing-Defaults) ---
        if db.query(FixedCost).count() == 0:
            defaults = [
                # name, amount_eur, vat_rate, cycle, skr03, eur_line
                ('Hetzner CX22 VPS',       4.00,  19.00, 'monthly', '4806', 52),
                ('nerve.app Domain',       1.25,  19.00, 'monthly', '4806', 57),
                ('Kontist Geschaeftskonto', 4.95, 19.00, 'monthly', '4970', 57),
                ('count.tax Steuerberater', 150.00, 19.00, 'monthly', '4950', 57),
                ('Homeoffice Tagespauschale', 6.00, 0.00, 'per_day', '4590', 65),
            ]
            for name, amt, vat, cycle, skr, line in defaults:
                db.add(FixedCost(name=name, amount_eur=amt, vat_rate=vat,
                                 cycle=cycle, skr03=skr, eur_line=line, active=True))
            print(f"[DB] Seeded {len(defaults)} fixed_costs (Phase 04.7.2)")

        # --- ApiRate Seed (Briefing Default-Preise in USD, werden per Wechselkurs umgerechnet) ---
        if db.query(ApiRate).count() == 0:
            rates = [
                # provider, model, unit_type, price, currency
                ('anthropic', 'haiku-4-5',    'per_1k_input_tokens',  0.00025, 'USD'),
                ('anthropic', 'haiku-4-5',    'per_1k_output_tokens', 0.00125, 'USD'),
                ('anthropic', 'sonnet-4',     'per_1k_input_tokens',  0.003,   'USD'),
                ('anthropic', 'sonnet-4',     'per_1k_output_tokens', 0.015,   'USD'),
                ('deepgram',  'nova-2',       'per_minute',           0.0036,  'USD'),
                ('elevenlabs','multilingual-v2','per_1k_chars',       0.30,    'USD'),
                ('stripe',    'card',         'percent',              0.014,   'EUR'),
                ('stripe',    'card',         'fixed_per_tx',         0.25,    'EUR'),
            ]
            for provider, model, unit_type, price, currency in rates:
                db.add(ApiRate(provider=provider, model=model, unit_type=unit_type,
                               price_per_unit=price, currency=currency, active=True,
                               source_url=f'briefing:2026-03-31/{provider}'))
            print(f"[DB] Seeded {len(rates)} api_rates (Phase 04.7.2)")

        # --- ExchangeRate Fallback-Seed (verhindert Divide-by-nothing im cost_tracker) ---
        if db.query(ExchangeRate).count() == 0:
            db.add(ExchangeRate(date=date.today(), currency_pair='USD_EUR',
                                rate=0.92, source='seed'))
            print("[DB] Seeded initial USD_EUR fallback rate=0.92 (Phase 04.7.2)")

        db.commit()
    except Exception as e:
        print(f"[DB] _seed_founder_dashboard_defaults failed: {e}")
        db.rollback()
    finally:
        db.close()
```

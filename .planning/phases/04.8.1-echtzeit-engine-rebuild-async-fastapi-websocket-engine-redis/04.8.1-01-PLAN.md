import logging
from contextlib import asynccontextmanager
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from nerve_rt.config import settings
from nerve_rt.redis_bridge import redis_bridge

logger = logging.getLogger("nerve_rt")

@asynccontextmanager
async def lifespan(app: FastAPI):
    # Startup
    await redis_bridge.connect()
    logger.info(f"[RT] Engine started on {settings.host}:{settings.port}")
    yield
    # Shutdown
    await redis_bridge.close()
    logger.info("[RT] Engine stopped")

app = FastAPI(
    title="NERVE Real-Time Engine",
    lifespan=lifespan,
)

app.add_middleware(
    CORSMiddleware,
    allow_origins=[settings.cors_origin] if settings.cors_origin != '*' else ['*'],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.get("/health")
async def health():
    redis_ok = await redis_bridge.ping()
    return {"status": "ok", "redis": redis_ok}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run("nerve_rt.main:app", host=settings.host, port=settings.port, reload=True)

Create `nerve_rt/redis_bridge.py` implementing the `RedisBridge` class:

```python
import json
import logging
from typing import Optional
from redis import asyncio as aioredis

logger = logging.getLogger("nerve_rt.redis")

# Redis key prefixes
SESSION_PREFIX = "nerve:session:"       # HSET — Flask writes, Engine reads
RESULTS_CHANNEL = "nerve:results:"      # PUB/SUB — Engine publishes, Flask subscribes
CONTROL_CHANNEL = "nerve:control:"      # PUB/SUB — Flask publishes, Engine subscribes

class RedisBridge:
    """Async Redis client for communication between Flask and RT Engine."""

    def __init__(self):
        self._redis: Optional[aioredis.Redis] = None
        self._pubsub: Optional[aioredis.client.PubSub] = None

    async def connect(self, url: str = None):
        from nerve_rt.config import settings
        redis_url = url or settings.redis_url
        self._redis = await aioredis.from_url(
            redis_url,
            decode_responses=True,
            max_connections=10,
        )
        logger.info(f"[Redis] Connected to {redis_url}")

    async def close(self):
        if self._pubsub:
            await self._pubsub.close()
        if self._redis:
            await self._redis.close()
        logger.info("[Redis] Connection closed")

    async def ping(self) -> bool:
        try:
            if self._redis:
                return await self._redis.ping()
            return False
        except Exception:
            return False

    # ── Session Token Operations ──────────────────────────────────────

    async def get_session(self, token: str) -> Optional[dict]:
        """Validate session token. Returns session data or None if expired/invalid.
        Flask writes: HSET nerve:session:{token} user_id X mode Y profile_id Z
        """
        if not self._redis:
            return None
        data = await self._redis.hgetall(f"{SESSION_PREFIX}{token}")
        if not data:
            return None
        # Delete token after read (single-use)
        await self._redis.delete(f"{SESSION_PREFIX}{token}")
        return data

    # ── Result Publishing (Engine -> Flask/Dashboard) ─────────────────

    async def publish_result(self, user_id: str, result: dict):
        """Publish analysis result for a user. Flask can subscribe for dashboard updates."""
        if not self._redis:
            return
        channel = f"{RESULTS_CHANNEL}{user_id}"
        await self._redis.publish(channel, json.dumps(result))

    # ── Control Subscription (Flask -> Engine) ────────────────────────

    async def subscribe_control(self, session_id: str):
        """Subscribe to control messages from Flask for a specific session."""
        if not self._redis:
            return None
        self._pubsub = self._redis.pubsub()
        await self._pubsub.subscribe(f"{CONTROL_CHANNEL}{session_id}")
        return self._pubsub

    async def listen_control(self, pubsub) -> dict:
        """Async generator for control messages. Use: async for msg in bridge.listen_control(ps)"""
        async for message in pubsub.listen():
            if message["type"] == "message":
                try:
                    yield json.loads(message["data"])
                except json.JSONDecodeError:
                    logger.warning(f"[Redis] Invalid control message: {message['data']}")

    # ── Store latest result for polling fallback ──────────────────────

    async def set_latest_result(self, user_id: str, result: dict, ttl: int = 300):
        """Store latest result in Redis hash for Flask polling fallback (5min TTL)."""
        if not self._redis:
            return
        key = f"nerve:latest:{user_id}"
        await self._redis.set(key, json.dumps(result), ex=ttl)

# Module-level singleton
redis_bridge = RedisBridge()
```

Key design decisions:
- `decode_responses=True` so all values come back as str (not bytes)
- Session token is single-use (deleted after read) to prevent replay attacks
- `max_connections=10` pool — sufficient for single-worker uvicorn
- Module-level singleton pattern (same as Flask `db` pattern in codebase)
- No audio bytes ever touch Redis (per D-09 / RESEARCH anti-pattern)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> Engine WS | Untrusted client connects to WebSocket endpoint |
| Flask -> Redis | Flask writes session tokens that Engine trusts |
| Redis -> Engine | Engine reads tokens and control messages from Redis |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04.8.1-01 | Spoofing | redis_bridge.get_session() | mitigate | Session tokens are single-use (deleted after first read), generated with secrets.token_urlsafe(32) on Flask side, 60s TTL in Redis |
| T-04.8.1-02 | Tampering | Redis channel messages | accept | Redis on localhost (127.0.0.1), no external access. Same trust level as Flask reading SQLite on same VPS |
| T-04.8.1-03 | Information Disclosure | config.py env vars | mitigate | API keys read from env vars (not hardcoded). .env file not committed to git (existing .gitignore rule) |
| T-04.8.1-04 | Denial of Service | Redis connection pool | mitigate | max_connections=10 pool limit prevents connection exhaustion |
</threat_model>

// ── Audio Manager (replaces Web Audio API oscillators) ─────────────────────
const AUDIO_UNIVERSAL = {
  klingeln:       '/static/audio/klingeln.mp3',
  besetztzeichen: '/static/audio/besetztzeichen.mp3',
  verbindungston: '/static/audio/verbindungston.mp3',
  wartemusik:     '/static/audio/wartemusik.mp3',
};
let _loopAudio = null;

function audioPlayLoop(src) {
  audioStop();
  _loopAudio = new Audio(src);
  _loopAudio.loop = true;
  _loopAudio.volume = 0.5;
  _loopAudio.play().catch(e => console.warn('[Audio]', e));
}

function audioPlayOnce(src) {
  return new Promise(resolve => {
    const a = new Audio(src);
    a.volume = 0.5;
    a.onended = resolve;
    a.onerror = resolve;
    a.play().catch(() => resolve());
  });
}

function audioStop() {
  if (_loopAudio) { _loopAudio.pause(); _loopAudio.currentTime = 0; _loopAudio = null; }
}

function startFreizeichen() {
  const lang = TRAINING_LANGUAGES[currentLanguage] || TRAINING_LANGUAGES['de'];
  const src = lang.audio_freizeichen || '/static/audio/freizeichen_de.mp3';
  audioPlayLoop(src);
}

function stopFreizeichen() {
  audioStop();
}
```

  await sleep(2500);

  overlay.style.display = 'none';

  // 4. Show popup with quick-stats per D-09
  showHangupPopup({
    grund: data.hangup_reason || 'Kunde hat aufgelegt',
    dauer: document.getElementById('t-timer') ? document.getElementById('t-timer').textContent : '—',
    turns: data.turn_count || 0,
    hat_sekretaerin: data.hat_sekretaerin || false,

function showHangupPopup(stats) {
  const title = (stats.hat_sekretaerin && !stats.sek_ueberwunden)
    ? 'Sekretaerin hat abgeblockt'
    : 'Anruf beendet';
  document.getElementById('t-hangup-title').textContent = title;
  let html = '<div style="display:grid; gap:6px;">';
  html += '<div><span style="color:#A0AEC0;">Grund:</span> ' + escH(stats.grund) + '</div>';
  html += '<div><span style="color:#A0AEC0;">Dauer:</span> ' + escH(stats.dauer) + '</div>';
  html += '<div><span style="color:#A0AEC0;">Nachrichten:</span> ' + stats.turns + '</div>';
  if (stats.hat_sekretaerin) {
    html += '<div><span style="color:#A0AEC0;">Sekretaerin ueberwunden:</span> ' +
      (stats.sek_ueberwunden ? '<span style="color:#22C55E;">Ja</span>' : '<span style="color:#EF4444;">Nein</span>') + '</div>';
  }
  html += '</div>';
  document.getElementById('t-hangup-stats').innerHTML = html;
  document.getElementById('t-hangup-popup').classList.remove('t-hidden');
}
```

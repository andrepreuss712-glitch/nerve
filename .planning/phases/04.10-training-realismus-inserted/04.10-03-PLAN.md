**Part D: JavaScript for Anruf-Typ selection**

Add JS variables and functions:
```javascript
let selectedAnrufTyp = 'direkt';
let selectedSekTyp   = 'blockerin';

function selectAnrufTyp(typ) {
  selectedAnrufTyp = typ;
  document.querySelectorAll('#t-anruftyp-section > .t-diff-grid > .t-diff-card').forEach(c => c.classList.remove('t-diff-active'));
  const card = document.querySelector('.t-diff-card[data-anruftyp="' + typ + '"]');
  if (card) card.classList.add('t-diff-active');
  // Show/hide sekretaerin type sub-cards
  document.getElementById('t-sektyp-cards').style.display = (typ === 'sekretaerin') ? '' : 'none';
}

function selectSekTyp(key, el) {
  selectedSekTyp = key;
  document.querySelectorAll('#t-sektyp-cards .t-diff-card').forEach(c => c.classList.remove('t-diff-active'));
  el.classList.add('t-diff-active');
}
```

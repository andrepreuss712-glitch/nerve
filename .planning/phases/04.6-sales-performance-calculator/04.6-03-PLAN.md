  function formatEuro(val) {
    if (val == null || isNaN(val)) return '—';
    val = Math.round(val);
    if (val >= 1000000) return (val / 1000000).toFixed(1).replace('.', ',') + ' Mio €';
    if (val >= 1000) return Math.round(val / 100) * 100 >= 1000
      ? (val / 1000).toFixed(0) + '.' + String(val % 1000).padStart(3, '0') + ' €'
      : val.toLocaleString('de-DE') + ' €';
    return val.toLocaleString('de-DE') + ' €';
  }

  function formatEuroShort(val) {
    if (val == null || isNaN(val)) return '—';
    val = Math.round(val);
    if (val >= 1000000) return (val/1000000).toFixed(1).replace('.', ',') + ' Mio €';
    if (val >= 1000) return Math.round(val/1000) + 'K €';
    return val + ' €';
  }

  function formatPercent(val, dec) {
    if (val == null || isNaN(val)) return '—';
    return val.toFixed(dec != null ? dec : 1).replace('.', ',') + '%';
  }

  // ── Simulations-Berechnung ────────────────────────────────────────────────────

  function perfSimUpdate() {
    // Label-Updates
    var callsTag  = parseInt(document.getElementById('sl-calls-tag').value);
    var showRate  = parseInt(document.getElementById('sl-show-rate').value);
    var closingCr = parseInt(document.getElementById('sl-closing-rate').value);
    var dealWert  = parseInt(document.getElementById('sl-deal-wert').value);
    var arbTage   = parseInt(document.getElementById('sl-arbeitstage').value);

    document.getElementById('lbl-calls-tag').textContent = callsTag;
    document.getElementById('lbl-show-rate').textContent = showRate + '%';
    document.getElementById('lbl-closing-rate').textContent = closingCr + '%';
    document.getElementById('lbl-deal-wert').textContent = formatEuroShort(dealWert);
    document.getElementById('lbl-arbeitstage').textContent = arbTage;

    // Debounce 150ms
    if (_simTimer) clearTimeout(_simTimer);
    _simTimer = setTimeout(function() {
      var callsProWoche = callsTag * arbTage;
      var callsProMonat = Math.round(callsProWoche * 4.3);
      var showsProMonat = Math.round(callsProMonat * showRate / 100);
      var dealsProMonat = Math.round(showsProMonat * closingCr / 100);
      var umsatzMonat   = dealsProMonat * dealWert;

      // Forecast mit 5% monatlichem Wachstum (S-Kurve)
      var f12 = 0;
      for (var i = 1; i <= 12; i++) {
        f12 += Math.round(umsatzMonat * Math.pow(1.05, i));
      }

      // ROI: 15% Einwand-Verbesserung Annahme
      var crDelta = closingCr * 0.15 / 100;
      var roiMehrwert = Math.round(callsProMonat * crDelta * dealWert) - 99;

      var simUmsatz = document.getElementById('sim-umsatz');
      var simF12    = document.getElementById('sim-forecast12');
      var simRoi    = document.getElementById('sim-roi');

      if (simUmsatz) simUmsatz.textContent = formatEuroShort(umsatzMonat);
      if (simF12)    simF12.textContent    = formatEuroShort(f12);
      if (simRoi) {
        simRoi.textContent = (roiMehrwert >= 0 ? '+' : '') + formatEuroShort(roiMehrwert);
        simRoi.style.color = roiMehrwert >= 0 ? '#00D4AA' : '#f87171';
      }
    }, 150);
  }

  // ── Deal-Wert Modal ───────────────────────────────────────────────────────────

  function openDealWertModal() {
    var modal = document.getElementById('deal-wert-modal');
    if (modal) modal.style.display = 'flex';
    setTimeout(function() {
      var input = document.getElementById('deal-wert-input');
      if (input) input.focus();

  function closeDealWertModal() {
    var modal = document.getElementById('deal-wert-modal');
    if (modal) modal.style.display = 'none';
  }

  function saveDealWert() {
    var input = document.getElementById('deal-wert-input');
    var err   = document.getElementById('deal-wert-error');
    var val   = parseInt(input ? input.value : '');
    if (!val || val < 1) {
      if (err) err.style.display = 'block';
      return;
    }
    if (err) err.style.display = 'none';

    fetch('/api/user/deal-wert', {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ avg_deal_wert: val }),
    })
    .then(function(r) { return r.json(); })
    .then(function(d) {
      if (d.ok) {
        closeDealWertModal();
        // Slider mit neuem Wert befuellen
        var slDw = document.getElementById('sl-deal-wert');
        if (slDw) {
          slDw.value = Math.min(50000, Math.max(100, val));
          document.getElementById('lbl-deal-wert').textContent = formatEuroShort(val);
        }
        // Performance neu laden
        perfInit();
      }
    })
    .catch(function() {});
  }

  // Enter-Taste im Modal
  document.addEventListener('DOMContentLoaded', function() {
    var inp = document.getElementById('deal-wert-input');
    if (inp) inp.addEventListener('keydown', function(e) { if (e.key === 'Enter') saveDealWert(); });
  });

  // ── Ergebnis-Tagging ──────────────────────────────────────────────────────────

  function tagSession(result) {
    if (!_tagSessionId) return;
    var bar = document.getElementById('result-tag-bar');
    var fb  = document.getElementById('tag-feedback');

    fetch('/api/session/' + _tagSessionId + '/result', {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ result: result }),
    })
    .then(function(r) { return r.json(); })
    .then(function(d) {

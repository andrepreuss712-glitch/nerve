    1. **routes/dashboard.py** — Bestehende `/analytics` Route umbauen (oder durch neue ersetzen):
       ```python
       @dashboard_bp.route('/analytics')
       @login_required
       def analytics():
           from database.models import ConversationLog
           db = get_session()
           try:
               mode = (request.args.get('mode') or '').strip()
               q = db.query(ConversationLog).filter(ConversationLog.user_id == g.user.id)
               if mode in ('cold_call','meeting','training'):
                   q = q.filter(ConversationLog.session_mode == mode)
               q = q.order_by(ConversationLog.id.desc()).limit(200)
               rows = q.all()
               return render_template('analytics.html',
                                      sessions=rows,
                                      active_mode=mode)
           finally:
               db.close()

       @dashboard_bp.route('/session/<int:sid>')
       @login_required
       def session_detail(sid):
           from database.models import ConversationLog, ObjectionEvent
           db = get_session()
           try:
               conv = db.query(ConversationLog).filter(
                   ConversationLog.id == sid,
                   ConversationLog.user_id == g.user.id,
               ).first()
               if not conv:
                   abort(404)
               events = db.query(ObjectionEvent).filter(
                   ObjectionEvent.conversation_log_id == sid,
               ).all()
               return render_template('session_detail.html', conv=conv, events=events)
           finally:
               db.close()
       ```
       Imports am Dateianfang sicherstellen (`abort` aus flask, `login_required`, `g`, `get_session`).

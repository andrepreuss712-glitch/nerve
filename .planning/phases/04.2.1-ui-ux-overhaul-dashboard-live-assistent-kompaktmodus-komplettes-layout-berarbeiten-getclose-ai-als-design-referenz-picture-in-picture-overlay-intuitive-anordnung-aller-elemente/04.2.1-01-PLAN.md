At the bottom of `base.html` (before `</body>`), add the sidebar toggle JS:
```javascript
function toggleSidebar() {
  var sb = document.querySelector('.g-sidebar');
  var collapsed = sb.classList.toggle('collapsed');
  try { localStorage.setItem('sn_sidebar_collapsed', collapsed ? '1' : '0'); } catch(e) {}
  // Update icon direction
  var icon = sb.querySelector('.sidebar-collapse-icon');
  if (icon) icon.setAttribute('data-lucide', collapsed ? 'chevrons-right' : 'chevrons-left');
  lucide.createIcons();
  // Update tooltip
  var btn = sb.querySelector('.sidebar-collapse-btn');
  if (btn) btn.title = collapsed ? 'Sidebar ausklappen' : 'Sidebar einklappen';
}
// Restore on load
(function() {
  try {
    if (localStorage.getItem('sn_sidebar_collapsed') === '1') {
      var sb = document.querySelector('.g-sidebar');
      if (sb) {
        sb.classList.add('collapsed');
        var icon = sb.querySelector('.sidebar-collapse-icon');
        if (icon) icon.setAttribute('data-lucide', 'chevrons-right');
      }
    }
  } catch(e) {}
})();
```

Call `lucide.createIcons()` at the end of the body script block to render all Lucide icons (it may already be called; verify and add if missing).

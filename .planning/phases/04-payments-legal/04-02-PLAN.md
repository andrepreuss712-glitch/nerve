    **1. Create fair-use utility function** — Add to `routes/payments.py` (or create `services/fair_use.py` if cleaner, Claude's discretion). The function:

    ```python
    def get_fair_use_status(org):
        """Return fair-use percentage and warning flags for an Organisation."""
        from datetime import datetime
        # Auto-reset if month changed (D-13)
        current_month = datetime.now().strftime('%Y-%m')
        if org.fair_use_reset_month != current_month:
            db = get_session()
            try:
                db.execute(
                    sa_update(Organisation)
                    .where(Organisation.id == org.id)
                    .values(
                        live_minutes_used=0,
                        training_sessions_used=0,
                        fair_use_reset_month=current_month,
                    )
                )
                db.commit()
            finally:
                db.close()
            # Reset in-memory too
            org.live_minutes_used = 0
            org.training_sessions_used = 0

        limit_min = max(org.minuten_limit or 1000, 1)
        limit_train = max(org.training_voice_limit or 50, 1)
        live_pct = min(100, round((org.live_minutes_used or 0) / limit_min * 100))
        train_pct = min(100, round((org.training_sessions_used or 0) / limit_train * 100))
        return {
            'live_pct': live_pct,
            'train_pct': train_pct,
            'live_used': org.live_minutes_used or 0,
            'live_limit': limit_min,
            'train_used': org.training_sessions_used or 0,
            'train_limit': limit_train,
            'warn_80': live_pct >= 80 or train_pct >= 80,
            'limit_reached': live_pct >= 100 or train_pct >= 100,
        }
    ```

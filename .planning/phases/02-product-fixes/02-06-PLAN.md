**JS:** Wizard navigation:
```javascript
let currentStep = 1;
const totalSteps = 3;

function nextStep() {
  // Validate current step
  if (currentStep === 1 && !document.getElementById('brancheInput').value) {
    alert('Waehle eine Branche um fortzufahren.');
    return;
  }
  if (currentStep === 3) return; // submit handles step 3
  document.getElementById('step-' + currentStep).style.display = 'none';
  currentStep++;
  document.getElementById('step-' + currentStep).style.display = 'block';
  updateNav();
  updateProgress();
}

function prevStep() {
  if (currentStep === 1) return;
  document.getElementById('step-' + currentStep).style.display = 'none';
  currentStep--;
  document.getElementById('step-' + currentStep).style.display = 'block';
  updateNav();
  updateProgress();
}

function updateNav() {
  document.getElementById('btnBack').style.display = currentStep > 1 ? '' : 'none';
  document.getElementById('btnNext').style.display = currentStep < totalSteps ? '' : 'none';
  document.getElementById('btnSubmit').style.display = currentStep === totalSteps ? '' : 'none';
}

function updateProgress() {
  for (let i = 1; i <= totalSteps; i++) {
    const dot = document.getElementById('dot-' + i);
    const line = document.getElementById('line-' + (i-1));
    dot.classList.toggle('active', i === currentStep);
    dot.classList.toggle('completed', i < currentStep);
    if (line) line.classList.toggle('completed', i <= currentStep);
  }
}

function selectBranche(branche) {
  document.querySelectorAll('.branche-card').forEach(c => c.classList.remove('selected'));
  event.currentTarget.classList.add('selected');
  document.getElementById('brancheInput').value = branche;
}

const selectedEinwaende = new Set();
function toggleEinwand(el, text) {
  el.classList.toggle('selected');
  if (selectedEinwaende.has(text)) selectedEinwaende.delete(text);
  else selectedEinwaende.add(text);
  document.getElementById('einwaendeInput').value = JSON.stringify([...selectedEinwaende]);
}
```
